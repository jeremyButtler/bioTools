define pseq
   set $pos = $arg0
   set $end = $arg1
   set $seqStr = $arg2

   while($pos <= $end)
      printf "%c", ($seqStr[$pos] + 64)
      set $pos = $pos + 1
   end

   printf "\n"
end

   printf "%u\t", samSTPtr->refStartUI + 1

end

define pkmers
   set $start = $arg0
   set $end = $arg1
   set $wrap = 0

   while($start <= $end)
      if($wrap >= 8)
          printf "\n"
          set $wrap = 0
      end

      if($wrap != 0)
         printf " "
      end

      set $index = refSTPtr->indexArySI[$start]
      printf "%i", refSTPtr->kmerArySI[$index]
      set $start = $start + 1
      set $wrap = $wrap + 1
  end

  printf "\n"
end
